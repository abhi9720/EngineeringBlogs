# JavaScript Prototype Chain: Complete Guide

## Overview

JavaScript uses prototypal inheritance—objects can inherit properties and methods from other objects. Understanding the prototype chain is essential for understanding how objects work in JavaScript, especially when using ES6 classes or creating custom data types.

---

## How the Prototype Chain Works

### Object Creation and Prototypes

Every JavaScript object (except null) has a hidden internal property called `[[Prototype]]` that links to another object—its prototype:

```javascript
const person = {
    name: 'John',
    greet() {
        return `Hello, ${this.name}`;
    }
};

console.log(person.name);  // 'John' - own property
console.log(person.greet()); // 'Hello, John' - from prototype

// Accessing prototype
console.log(Object.getPrototypeOf(person));  // Object.prototype

// The prototype chain: person → Object.prototype → null
```

### Constructor Functions

```javascript
// Constructor function - creates objects with shared methods
function Person(name, age) {
    this.name = name;
    this.age = age;
}

// Adding method to prototype - shared by all instances
Person.prototype.greet = function() {
    return `Hello, I'm ${this.name}`;
};

Person.prototype.getAge = function() {
    return `${this.name} is ${this.age} years old`;
};

// Creating instances
const john = new Person('John', 30);
const jane = new Person('Jane', 25);

console.log(john.greet());  // "Hello, I'm John"
console.log(jane.greet());  // "Hello, I'm Jane"

// Both share the same method from prototype!
console.log(john.greet === jane.greet);  // true
```

## Real-World JavaScript Use Cases

### Case 1: Custom Data Structures

```javascript
// Creating a custom list implementation
function LinkedList() {
    this.head = null;
    this.length = 0;
}

LinkedList.prototype.push = function(value) {
    const node = { value, next: null };

    if (!this.head) {
        this.head = node;
    } else {
        let current = this.head;
        while (current.next) {
            current = current.next;
        }
        current.next = node;
    }

    this.length++;
    return this;
};

LinkedList.prototype.pop = function() {
    if (!this.head) return null;

    let current = this.head;
    let previous = null;

    while (current.next) {
        previous = current;
        current = current.next;
    }

    if (previous) {
        previous.next = null;
    } else {
        this.head = null;
    }

    this.length--;
    return current.value;
};

LinkedList.prototype.getAt = function(index) {
    if (index < 0 || index >= this.length) return null;

    let current = this.head;
    for (let i = 0; i < index; i++) {
        current = current.next;
    }
    return current.value;
};

// Usage
const list = new LinkedList();
list.push(1).push(2).push(3);
console.log(list.getAt(1));  // 2
console.log(list.pop());     // 3
```

### Case 2: Inheriting from Built-in Types

```javascript
// Extending Array with custom methods
class Stack extends Array {
    constructor() {
        super();
    }

    push(item) {
        super.push(item);
        console.log(`Pushed: ${item}`);
        return this;
    }

    pop() {
        const item = super.pop();
        console.log(`Popped: ${item}`);
        return item;
    }

    peek() {
        return this[this.length - 1];
    }

    isEmpty() {
        return this.length === 0;
    }
}

const stack = new Stack();
stack.push(1).push(2).push(3);
console.log(stack.peek());  // 3
stack.pop();  // Popped: 3
console.log(stack);  // [1, 2]

// Another example: Custom string methods
String.prototype.capitalize = function() {
    return this.charAt(0).toUpperCase() + this.slice(1);
};

String.prototype.titleCase = function() {
    return this.split(' ')
        .map(word => word.charAt(0).toUpperCase() + word.slice(1))
        .join(' ');
};

console.log('hello world'.capitalize());  // "Hello world"
console.log('hello world'.titleCase());   // "Hello World"
```

### Case 3: Mixins (Multiple Inheritance)

```javascript
// Mixin - adds properties/methods from one object to another
const canWalk = {
    walk() {
        return `${this.name} is walking`;
    }
};

const canSwim = {
    swim() {
        return `${this.name} is swimming`;
    }
};

const canFly = {
    fly() {
        return `${this.name} is flying`;
    }
};

// Creating an amphibian
function createAmphibian(name) {
    const amphibian = { name };
    Object.assign(amphibian, canWalk, canSwim);
    return amphibian;
}

const frog = createAmphibian('Frog');
console.log(frog.walk());  // "Frog is walking"
console.log(frog.swim());  // "Frog is swimming"

// Mixin with ES6 classes
class Duck {
    constructor(name) {
        this.name = name;
    }
}

// Apply mixins
Object.assign(Duck.prototype, canWalk, canSwim, canFly);

const duck = new Duck('Donald');
console.log(duck.walk());  // "Donald is walking"
console.log(duck.swim());  // "Donald is swimming"
console.log(duck.fly());   // "Donald is flying"

// Functional mixins
const withLogging = fn => function(...args) {
    console.log(`Calling ${fn.name} with:`, args);
    return fn.apply(this, args);
};

function createLogger(target) {
    return new Proxy(target, {
        get(obj, prop) {
            const value = obj[prop];
            if (typeof value === 'function') {
                return withLogging(value);
            }
            return value;
        }
    });
}

const loggedObj = createLogger({ add(a, b) { return a + b; } });
loggedObj.add(1, 2);  // Logs: Calling add with: [1, 2]
```

### Case 4: Object.create for Inheritance

```javascript
// Using Object.create for inheritance without constructor functions
const animal = {
    speak() {
        return `${this.name} makes a sound`;
    },
    eat() {
        return `${this.name} is eating`;
    }
};

const dog = Object.create(animal);
dog.name = 'Buddy';
dog.bark = function() {
    return `${this.name} barks`;
};

console.log(dog.speak());  // "Buddy makes a sound"
console.log(dog.bark());   // "Buddy barks"

// Check prototype chain
console.log(Object.getPrototypeOf(dog) === animal);  // true
console.log(Object.getPrototypeOf(animal) === Object.prototype);  // true

// Another example: Customizable objects
const withDefaults = (defaults) => (props) => ({ ...defaults, ...props });

const createUser = withDefaults({
    role: 'user',
    active: true,
    theme: 'light'
});

const user1 = createUser({ name: 'John' });
console.log(user1);  // { name: 'John', role: 'user', active: true, theme: 'light' }
```

## ES6 Classes and Prototypes

### Classes Are Syntactic Sugar

```javascript
// ES6 class syntax
class Person {
    constructor(name, age) {
        this.name = name;
        this.age = age;
    }

    greet() {
        return `Hello, I'm ${this.name}`;
    }

    static create(name) {
        return new this(name, 0);
    }
}

// Under the hood, this creates the same prototype chain as:
function Person(name, age) {
    this.name = name;
    this.age = age;
}
Person.prototype.greet = function() {
    return `Hello, I'm ${this.name}`;
};

// Verify they're the same
const person = new Person('John', 30);
console.log(person.greet());
console.log(Object.getPrototypeOf(person).constructor === Person);
```

### Class Inheritance

```javascript
class Animal {
    constructor(name) {
        this.name = name;
    }

    speak() {
        return `${this.name} makes a sound`;
    }
}

class Dog extends Animal {
    constructor(name, breed) {
        super(name);  // Call parent constructor
        this.breed = breed;
    }

    speak() {
        return `${this.name} barks`;
    }

    fetch() {
        return `${this.name} fetches the ball`;
    }
}

const buddy = new Dog('Buddy', 'Golden Retriever');
console.log(buddy.speak());   // "Buddy barks"
console.log(buddy.fetch());   // "Buddy fetches the ball"
console.log(buddy instanceof Dog);     // true
console.log(buddy instanceof Animal);  // true
```

### Private Fields with Symbols

```javascript
// Using Symbols for private properties
const _name = Symbol('name');
const _age = Symbol('age');

class Person {
    constructor(name, age) {
        this[_name] = name;
        this[_age] = age;
    }

    getName() {
        return this[_name];
    }

    getAge() {
        return this[_age];
    }
}

const person = new Person('John', 30);
console.log(person.getName());  // 'John'
console.log(person._name);      // undefined (private!)
console.log(Object.getOwnPropertySymbols(person));  // [Symbol(name), Symbol(age)]
```

## Production Considerations

### 1. Performance Implications

```javascript
// Prototype chain lookup is fast but has overhead
// Avoid deep prototype chains for performance

// Bad: Deep chain
function Level1() {}
Level1.prototype.method = () => 'level1';

function Level2() {}
Level2.prototype = Object.create(Level1.prototype);
Level2.prototype.method = () => 'level2';

function Level3() {}
Level3.prototype = Object.create(Level2.prototype);
Level3.prototype.method = () => 'level3';

// Good: Flat structure with composition
class Level3 {
    constructor() {
        this.level1 = new Level1();
        this.level2 = new Level2();
    }

    method() {
        // Delegate to appropriate instance
        return this.level2.method();
    }
}
```

### 2. Memory Efficiency

```javascript
// Methods on prototype vs on instance
function BadPattern(name) {
    this.name = name;
    this.greet = function() {  // New function for EACH instance!
        return `Hello, ${this.name}`;
    };
}

function GoodPattern(name) {
    this.name = name;
}

GoodPattern.prototype.greet = function() {  // Single function shared
    return `Hello, ${this.name}`;
};

const p1 = new BadPattern('John');
const p2 = new BadPattern('Jane');
console.log(p1.greet === p2.greet);  // false - different functions!

const g1 = new GoodPattern('John');
const g2 = new GoodPattern('Jane');
console.log(g1.greet === g2.greet);  // true - same function!
```

## Common Mistakes

### Mistake 1: Modifying Object.prototype

```javascript
// WRONG: Polluting the global prototype
Object.prototype.getKeys = function() {
    return Object.keys(this);
};

const obj = { a: 1, b: 2 };
console.log(obj.getKeys());  // Works but pollutes ALL objects!

// CORRECT: Create custom methods on specific objects or classes
function createExtendedObject(obj) {
    return {
        ...obj,
        getKeys() {
            return Object.keys(this);
        }
    };
}
```

### Mistake 2: Not Using .call() or .bind() for Methods

```javascript
// WRONG: Losing 'this' context
const person = {
    name: 'John',
    greet() {
        return `Hello, ${this.name}`;
    }
};

const greetFn = person.greet;
console.log(greetFn());  // "Hello, undefined" - 'this' is lost!

// CORRECT: Bind the method
const boundGreet = person.greet.bind(person);
console.log(boundGreet());  // "Hello, John"

// Or use arrow function in the object
const person2 = {
    name: 'John',
    greet: () => `Hello, ${this.name}`  // Arrow functions don't have their own 'this'
};
console.log(person2.greet());  // "Hello, undefined"
```

### Mistake 3: Confusing __proto__ and prototype

```javascript
// __proto__ is the prototype of an object (the internal [[Prototype]])
// prototype is a property used with constructor functions

function Person(name) {
    this.name = name;
}
Person.prototype.greet = function() { return `Hello, ${this.name}`; };

const john = new Person('John');

// __proto__ - the object's prototype chain
console.log(john.__proto__ === Person.prototype);  // true

// prototype - used by 'new' to set __proto__
console.log(Person.prototype.greet.call(john));  // Works!

console.log(john.prototype);  // undefined! Objects don't have .prototype
console.log(Person.prototype);  // The prototype object
```

### Mistake 4: Not Calling super()

```javascript
// WRONG: Forgetting to call super() in constructor
class Child extends Parent {
    constructor(name, age) {
        // Must call super() before accessing 'this'
        // this.age = age;  // ReferenceError!

        super(name);  // Must be first!
        this.age = age;
    }
}

// CORRECT: Always call super() first
class Parent {
    constructor(name) {
        this.name = name;
    }
}

class Child extends Parent {
    constructor(name, age) {
        super(name);  // Call parent constructor
        this.age = age;
    }
}
```

### Mistake 5: Not Checking instanceof

```javascript
// Using instanceof for type checking
class Animal {
    constructor(name) {
        this.name = name;
    }
}

class Dog extends Animal {
    bark() {
        return 'Woof!';
    }
}

const dog = new Dog('Buddy');

console.log(dog instanceof Dog);     // true
console.log(dog instanceof Animal);   // true
console.log(dog instanceof Object);  // true

// Be careful with cross-frame comparisons
// Different iframe has different Array constructor
// Use Array.isArray() instead of instanceof
```

## Summary

Understanding the prototype chain is fundamental to JavaScript:

1. **Prototype**: Every object has a prototype link (except null)
2. **Property Lookup**: JavaScript walks the prototype chain to find properties
3. **Constructor Functions**: Set up prototype chain with `.prototype`
4. **ES6 Classes**: Syntactic sugar over prototypal inheritance

Key insights:
- Methods on prototype are shared across all instances (memory efficient)
- Use `Object.create()` for custom inheritance without constructors
- Classes extend other classes using `extends` and `super()`
- Private fields can use Symbols or # prefix (ES2022)

Understanding prototypes helps you make better architectural decisions and debug issues faster.
